Variables

Constant Variables:
	const NAME = VALUE

Data Types:
  -  bool	1 bit		0 = false, 1 = true
  -  int	automatically set to 32 or 64 bit
  -  uint		"		"		"		"
  -  float32, float46
  -  string
  
Declaration and Initialization:
		Explicit data type:
	var NAME TYPE	,	NAME = VALUE
	var NAME TYPE = VALUE
		Inferred data type:
	var NAME = VALUE
	NAME := VALUE
	
Multiple Declarations:
	thing1, thing2 = 56, "stuff"
	-> thing1 int = 56, thing2 string = "stuff
